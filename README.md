## PASOS PARA LEVANTAR TODO EL PROYECTO

# Antes de todo
# 1. cp .env.example .env

# En una terminal

# 1. cd backend
# 2. pyenv install 3.13.2 (INSTALAR pyenv)
# 3. pyenv virtualenv 3.13.2 number-converter-env
# 4. pyenv activate number-converter-env
# 5. pip install -r --upgrade pip
# 6. pip install -r requirements.txt
# 7. python -m uvicorn app.main:app --reload

# En otra terminal (sin cerrar la anterior)

# 1. npm install
# 2. npm run dev
